  }

  const first = input.indexOf("'");

function candidate(input) {
  if (typeof input !== "string" || input.length === 0) {
    throw new TypeError("invalid content disposition");
  }

  var token = /^[!#$%&'*+.^_`|~0-9A-Za-z-]+/;
  var first = token.exec(input);
  if (!first || first.index !== 0) {
    throw new TypeError("invalid content disposition");
  }

  var type = first[0].toLowerCase();
  var pos = first[0].length;
  var params = {};
  var seen = Object.create(null);
  var extended = Object.create(null);

  function skipSpace(i) {
    while (i < input.length) {
      var c = input.charCodeAt(i);
      if (c !== 32 && c !== 9) break;
      i++;
    }
    return i;
  }

  function bad() {
    throw new TypeError("invalid content disposition");
  }

  function readToken(i) {
    var m = token.exec(input.slice(i));
    return m ? m[0] : "";
  }

  function decodeExtended(text) {
    var pieces = /^([^']*)'[^']*'(.*)$/.exec(text);
    if (!pieces) bad();

    var charset = pieces[1].toLowerCase();
    var body = pieces[2];
    var bytes = [];

    for (var i = 0; i < body.length; i++) {
      var ch = body.charCodeAt(i);
      if (body[i] === "%") {
        var hex = body.slice(i + 1, i + 3);
        if (!/^[0-9A-Fa-f]{2}$/.test(hex)) bad();
        bytes.push(parseInt(hex, 16));
        i += 2;
      } else {
        bytes.push(ch);
      }
    }

    if (charset === "utf-8" || charset === "utf8") {
      if (typeof TextDecoder !== "undefined") {
        return new TextDecoder("utf-8", { fatal: true }).decode(new Uint8Array(bytes));
      }
      return decodeURIComponent(body.replace(/%(?![0-9A-Fa-f]{2})/g, "%25"));
    }

    if (charset === "iso-8859-1" || charset === "latin1") {
      var out = "";
      for (var j = 0; j < bytes.length; j++) out += String.fromCharCode(bytes[j]);
      return out;
    }

    bad();
  }

  pos = skipSpace(pos);
  if (pos === input.length) return { type: type, parameters: params };
  if (input[pos] !== ";") bad();

  while (pos < input.length) {
    if (input[pos] !== ";") bad();

    pos = skipSpace(pos);
    if (pos === input.length) break;

    var rawName = readToken(pos);
    if (!rawName) bad();
    pos += rawName.length;

          if (pos >= input.length) bad();
          rawValue += input[pos++];
        } else {
          rawValue += chr;
        }
      }
      if (input[pos - 1] !== '"') bad();
    } else {
      rawValue = readToken(pos);
      if (!rawValue) bad();
      pos += rawValue.length;
    }

    pos = skipSpace(pos);
    if (pos < input.length && input[pos] !== ";") bad();

    var lowered = rawName.toLowerCase();
    if (seen[lowered]) {
      throw new TypeError("duplicate parameter");
    }
    seen[lowered] = true;

    var isExtended = lowered.endsWith("*");

    if (!name) bad();

    if (isExtended) {
      params[name] = decodeExtended(rawValue);
      extended[name] = true;
    } else if (!extended[name]) {
      params[name] = rawValue;
    }
  }

  return { type: type, parameters: params };
}
